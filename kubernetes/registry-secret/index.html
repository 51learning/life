    #!/bin/bash  
ns_list=`kubectl get ns | awk '{print $1}' | grep -v NAME`  
for ns in $ns_list;  
do  
kubectl create secret docker-registry imagePullSecret-registry   
--docker-server=服务器地址   
--docker-username=admin   
--docker-password=******   
--docker-email=******   
-n $ns  
done;
